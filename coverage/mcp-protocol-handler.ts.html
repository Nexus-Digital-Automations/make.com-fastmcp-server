/**
 * MCP Protocol Message Handler
 * Provides robust JSON parsing and error handling for MCP protocol messages
 */
 
import logger from './logger.js';
 
const getComponentLogger = (): ReturnType<typeof logger.child> => {
    try {
      return logger.child({ component: 'MCPProtocolHandler' });
    } catch {
      // Fallback for test environments
      // eslint-disable-next-line @typescript-eslint/no-explicit-any
      return logger as any;
    }
  };
  const componentLogger = getComponentLogger();
 
export interface MCPMessage {
  jsonrpc: string;
  id?: string | number;
  method?: string;
  params?: unknown;
  result?: unknown;
  error?: {
    code: number;
    message: string;
    data?: unknown;
  };
}

export class MCPProtocolHandler {
  /**
   * Safely parse JSON with enhanced error handling for MCP messages
   */
  static safeParseJSON(data: string): MCPMessage | null {
    try {
      // Clean up common JSON formatting issues
      const cleanedData = data.trim();
      
      // Check for incomplete JSON (common cause of parsing errors)
      if (!cleanedData || cleanedData.length < 2) {
        componentLogger.warn('Received empty or too short JSON data', { 
          dataLength: cleanedData.length,
          data: cleanedData.substring(0, 50) 
        });
        return null;
      }
 
      // Check for basic JSON structure
      if (!cleanedData.startsWith('{') && !cleanedData.startsWith('[')) {
        componentLogger.warn('JSON data does not start with valid character', { 
          firstChar: cleanedData[0],
          data: cleanedData.substring(0, 50) 
        });
        return null;
      }
 
      // Check for matching braces/brackets
      if (cleanedData.startsWith('{') && !cleanedData.endsWith('}')) {
        componentLogger.warn('JSON object appears incomplete', { 
          data: cleanedData.substring(0, 50) + '...' + cleanedData.substring(-10) 
        });
        return null;
      }
 
      if (cleanedData.startsWith('[') && !cleanedData.endsWith(']')) {
        componentLogger.warn('JSON array appears incomplete', { 
          data: cleanedData.substring(0, 50) + '...' + cleanedData.substring(-10) 
        });
        return null;
      }

      const parsed = JSON.parse(cleanedData);
      
      // Validate MCP message structure
      if (typeof parsed === 'object' && parsed !== null) {
        const message = parsed as MCPMessage;
        
        // Check for required jsonrpc field
        if (!message.jsonrpc) {
          componentLogger.warn('MCP message missing jsonrpc field', { parsed });
          return null;
        }
 
        // Validate JSON-RPC 2.0 compliance
        if (message.jsonrpc !== '2.0') {
          componentLogger.warn('Invalid JSON-RPC version', { 
            version: message.jsonrpc,
            expected: '2.0' 
          });
        }

        return message;
      }

      componentLogger.warn('Parsed JSON is not an object', { 
        type: typeof parsed,
        parsed 
      });
      return null;

    } catch (error) {
      if (error instanceof SyntaxError) {
        componentLogger.error('JSON parsing error', {
          message: error.message,
          position: this.extractErrorPosition(error.message),
          data: data.substring(0, 100) + (data.length > 100 ? '...' : ''),
          dataLength: data.length
        });
      } else {
        componentLogger.error('Unexpected error during JSON parsing', {
          error: error instanceof Error ? error.message : String(error),
          data: data.substring(0, 50)
        });
      }
      return null;
    }
  }

  /**
   * Extract error position from JSON syntax error message
   */
  private static extractErrorPosition(errorMessage: string): number | null {
    const positionMatch = errorMessage.match(/position (\d+)/);
    return positionMatch ? parseInt(positionMatch[1], 10) : null;
  }

  /**
   * Create error response for malformed messages
   */
  static createErrorResponse(id: string | number | undefined, error: {
    code: number;
    message: string;
    data?: unknown;
  }): MCPMessage {
    return {
      jsonrpc: '2.0',
      id: id,
      error
    };
  }
 
  /**
   * Handle protocol-level errors with detailed logging
   */
  static handleProtocolError(data: string, error: Error): void {
    componentLogger.error('MCP protocol error', {
      errorType: error.constructor.name,
      errorMessage: error.message,
      dataPreview: data.substring(0, 100),
      dataLength: data.length,
      stack: error.stack
    });

    // Log additional context for common errors
    if (error.message.includes('Unexpected end of input')) {
      componentLogger.error('Possible incomplete message transmission', {
        suggestion: 'Check message framing and buffer handling'
      });
    }

    if (error.message.includes('Expected')) {
      const position = this.extractErrorPosition(error.message);
      if (position !== null && position < data.length) {
        componentLogger.error('JSON syntax error context', {
          position,
          character: data[position],
          context: data.substring(Math.max(0, position - 10), position + 10)
        });
      }
    }
  }
}

export default MCPProtocolHandler;
